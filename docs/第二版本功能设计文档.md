### 2.1 智能推荐系统

#### 2.1.1 功能描述
基于用户历史行为、偏好和位置信息，提供个性化的景点、美食、住宿推荐。

#### 2.1.2 技术实现
```python
# 推荐系统核心代码
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity
from typing import List, Dict

class RecommendationSystem:
    def __init__(self, user_preferences: Dict):
        self.user_preferences = user_preferences
        self.user_item_matrix = self._build_user_item_matrix()
        self.user_similarity = cosine_similarity(self.user_item_matrix)
    
    def _build_user_item_matrix(self) -> np.ndarray:
        # 构建用户-物品矩阵
        users = list(self.user_preferences.keys())
        items = self._get_all_items()
        matrix = np.zeros((len(users), len(items)))
        for i, user in enumerate(users):
            for j, item in enumerate(items):
                matrix[i][j] = self._get_user_item_score(user, item)
        return matrix
    
    def recommend(self, user_id: str, top_n: int = 10) -> List[Dict]:
        # 获取用户相似度
        user_idx = self._get_user_index(user_id)
        similarities = self.user_similarity[user_idx]
        # 计算加权评分
        scores = np.dot(similarities, self.user_item_matrix) / np.sum(similarities)
        # 获取未评分项目
        user_ratings = self.user_item_matrix[user_idx]
        unrated_items = np.where(user_ratings == 0)[0]
        # 排序并返回Top N
        recommended_items = sorted(
            [(self._get_item_by_index(item_idx), scores[item_idx]) 
             for item_idx in unrated_items],
            key=lambda x: x[1],
            reverse=True
        )[:top_n]
        return recommended_items
```

```python
# 预算管理核心代码
from pydantic import BaseModel
from typing import List, Optional
from datetime import datetime

class BudgetCategory(BaseModel):
    name: str
    budget: float
    spent: float = 0.0

class Budget(BaseModel):
    budget_id: str
    trip_id: str
    total_budget: float
    categories: List[BudgetCategory]
    created_at: datetime = datetime.now()
    updated_at: datetime = datetime.now()

class BudgetService:
    def create_budget(self, trip_id: str, total_budget: float, categories: List[Dict]) -> Budget:
        # 创建预算
        budget_categories = [BudgetCategory(**cat) for cat in categories]
        budget = Budget(
            budget_id=f"budget_{datetime.now().timestamp()}",
            trip_id=trip_id,
            total_budget=total_budget,
            categories=budget_categories
        )
        # 保存到数据库
        self._save_budget(budget)
        return budget
    
    def add_expense(self, budget_id: str, category: str, amount: float, description: str = "") -> Budget:
        # 添加消费记录
        budget = self._get_budget(budget_id)
        for cat in budget.categories:
            if cat.name == category:
                cat.spent += amount
                break
        budget.updated_at = datetime.now()
        self._save_budget(budget)
        return budget
```

```python
import zipfile
import os
from typing import Dict

class OfflineService:
    def __init__(self, offline_dir: str = "offline_data"):
        self.offline_dir = offline_dir
        os.makedirs(offline_dir, exist_ok=True)
    
    def download_map(self, city: str) -> str:
        # 下载离线地图
        map_file = f"{self.offline_dir}/{city}_map.zip"
        # 生成地图数据包
        self._generate_map_package(city, map_file)
        return map_file
    
    def _generate_map_package(self, city: str, output_file: str) -> None:
        # 生成地图数据包
        with zipfile.ZipFile(output_file, 'w') as zipf:
            # 添加地图数据文件
            map_data = self._get_map_data(city)
            zipf.writestr(f"{city}_map.json", map_data)
            # 添加元数据
            metadata = {"version": "1.0", "city": city, "created_at": datetime.now().isoformat()}
            zipf.writestr("metadata.json", json.dumps(metadata))
    
    def sync_offline_data(self, user_id: str, data: Dict) -> None:
        # 同步离线数据
        user_dir = f"{self.offline_dir}/{user_id}"
        os.makedirs(user_dir, exist_ok=True)
        sync_file = f"{user_dir}/sync_{datetime.now().timestamp()}.json"
        with open(sync_file, 'w') as f:
            json.dump(data, f)
```

```python
# 语音交互核心代码
import speech_recognition as sr
from gtts import gTTS
import os
from typing import Dict

class VoiceService:
    def __init__(self, language: str = "zh-CN"):
        self.language = language
        self.recognizer = sr.Recognizer()
    
    def recognize_voice(self, voice_data: str) -> str:
        # 语音识别
        audio_file = sr.AudioFile(voice_data)
        with audio_file as source:
            audio = self.recognizer.record(source)
        try:
            text = self.recognizer.recognize_google(audio, language=self.language)
            return text
        except sr.UnknownValueError:
            return "无法识别语音"
        except sr.RequestError:
            return "语音识别服务不可用"
    
    def synthesize_voice(self, text: str) -> str:
        # 语音合成
        tts = gTTS(text=text, lang=self.language)
        audio_file = f"audio_{datetime.now().timestamp()}.mp3"
        tts.save(audio_file)
        return audio_file
    
    def process_voice_command(self, voice_data: str) -> Dict:
        # 处理语音指令
        text = self.recognize_voice(voice_data)
        intent = self._parse_intent(text)
        parameters = self._extract_parameters(text, intent)
        response = self._generate_response(intent, parameters)
        return {
            "intent": intent,
            "parameters": parameters,
            "response": response
        }
```

```python
# Redis缓存优化
import redis
import json
from functools import wraps

redis_client = redis.Redis(host='localhost', port=6379, db=0)

def cache_result(expire_time: int = 3600):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            cache_key = f"{func.__name__}:{args}:{kwargs}"
            cached_result = redis_client.get(cache_key)
            if cached_result:
                return json.loads(cached_result)
            result = func(*args, **kwargs)
            redis_client.setex(cache_key, expire_time, json.dumps(result))
            return result
        return wrapper
    return decorator

# 使用缓存装饰器
@cache_result(expire_time=3600)

    # 推荐系统逻辑
```

```python
# RabbitMQ异步任务处理
import pika
import json
from typing import Dict

class AsyncTaskManager:
    def __init__(self, rabbitmq_host: str = 'localhost'):
        self.connection = pika.BlockingConnection(pika.ConnectionParameters(rabbitmq_host))
        self.channel = self.connection.channel()
        self.channel.queue_declare(queue='trip_planning', durable=True)
    
    def send_task(self, task_type: str, task_data: Dict) -> None:
        # 发送异步任务
        self.channel.basic_publish(
            exchange='',
            routing_key='trip_planning',
            body=json.dumps({"type": task_type, "data": task_data}),
            properties=pika.BasicProperties(
                delivery_mode=2,  # 持久化消息
            )
        )
    
    def process_tasks(self, callback):
        # 处理异步任务
        self.channel.basic_qos(prefetch_count=1)
        self.channel.basic_consume(
            queue='trip_planning',
            on_message_callback=callback
        )
        self.channel.start_consuming()
```

## 🛡️ 安全与隐私设计

### 5.1 数据加密
```python
# 数据加密核心代码
import hashlib
import base64
from cryptography.fernet import Fernet

class SecurityService:
    def __init__(self, encryption_key: str):
        self.fernet = Fernet(encryption_key)
    
    def encrypt_data(self, data: str) -> str:
        # 加密数据
        encrypted_data = self.fernet.encrypt(data.encode())
        return base64.b64encode(encrypted_data).decode()
    
    def decrypt_data(self, encrypted_data: str) -> str:
        # 解密数据
        decoded_data = base64.b64decode(encrypted_data)
        decrypted_data = self.fernet.decrypt(decoded_data)
        return decrypted_data.decode()
    
    def hash_password(self, password: str) -> str:
        # 密码哈希
        salt = os.urandom(16)
        hashed_password = hashlib.pbkdf2_hmac(
            'sha256',
            password.encode('utf-8'),
            salt,
            100000
        )
        return base64.b64encode(salt + hashed_password).decode()
```

```python
# JWT身份认证
import jwt
from datetime import datetime, timedelta
from typing import Dict

class AuthService:
    def __init__(self, secret_key: str, algorithm: str = "HS256"):
        self.secret_key = secret_key
        self.algorithm = algorithm
    
    def create_token(self, user_id: str, expires_delta: timedelta = None) -> str:
        # 创建JWT Token
        to_encode = {"sub": user_id, "exp": datetime.utcnow() + (expires_delta or timedelta(days=7))}
        encoded_jwt = jwt.encode(to_encode, self.secret_key, algorithm=self.algorithm)
        return encoded_jwt
    
    def verify_token(self, token: str) -> Dict:
        # 验证JWT Token
        try:
            payload = jwt.decode(token, self.secret_key, algorithms=[self.algorithm])
            return payload
        except jwt.ExpiredSignatureError:
            raise Exception("Token已过期")
        except jwt.InvalidTokenError:
            raise Exception("无效的Token")
```
